153class event_t {
154private:
155 uint8_t UdId; // UpDown ID
156 uint8_t LaneId; // Lane ID
157 uint8_t ThreadId; // Thread ID. ANY_THREAD for any thread
158 uint8_t EventLabel; // Number representing the event label
159
160 operands_t *Operands; // Operands to be sent with this event
161
162 word_t EventWord; // Concatenated event word that contains all the fields
163
164public:
169 event_t()
170 : UdId(0), LaneId(0), ThreadId(0xFF), EventLabel(0), Operands(nullptr),
171 EventWord(((LaneId << 24) & 0xff000000) |
172 ((ThreadId << 16) & 0xff0000) | ((0 << 8) & 0xff00) |
173 (EventLabel & 0xff)) {
174 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
175 "thread_id = %d, num_operands = %d, ev_word = %X",
176 EventLabel, LaneId, ThreadId, 0, EventWord);
177 }
178
188 event_t(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
189 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr)
190 : UdId(udid), LaneId(lid), ThreadId(tid), EventLabel(e_label),
191 Operands(operands),
192 EventWord(
193 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
194 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
195 0xff00) |
196 (EventLabel & 0xff)) {
197 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
198 "thread_id = %d, num_operands = %d, ev_word = %X",
199 EventLabel, LaneId, ThreadId,
200 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
201 EventWord);
202 }
203
212 void set_event(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
213 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr) {
214 EventLabel = e_label;
215 LaneId = lid;
216 UdId = udid;
217 ThreadId = tid;
218 Operands = operands;
219 EventWord =
220 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
221 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
222 0xff00) |
223 (EventLabel & 0xff);
224 UPDOWN_INFOMSG("Setting the event word = %d, lane_id = %d, "
225 "thread_id = %d, num_operands = %d, ev_word = %X",
226 EventLabel, LaneId, ThreadId,
227 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
228 EventWord);
229 }
230
231 word_t get_EventWord() { return EventWord; }
232 operands_t *get_Operands() { return Operands; }
233 void set_operands(operands_t *ops) { Operands = ops; }
234 uint8_t get_UdId() { return UdId; }
235 uint8_t get_LaneId() { return LaneId; }
236 uint8_t get_ThreadId() { return ThreadId; }
237 uint8_t get_EventLabel() { return EventLabel; }
238 uint8_t get_NumOperands() {
239 return (Operands != nullptr) ? Operands->get_NumOperands() : 0;
240 }
241 ptr_t get_OperandsData() {
242 return (Operands != nullptr) ? Operands->get_Data() : nullptr;
243 }

437 class ud_mapped_memory_t {
438 private:
445 struct mem_region_t {
446 uint64_t size; // Size of the region
447 bool free; // Is the region free or being used
448 };
449
450 // Map to keep track of the free and used regions.
451 std::map<void *, mem_region_t> regions;
452
453 public:
462 ud_mapped_memory_t(ud_machine_t &machine) {
463 UPDOWN_INFOMSG("Initializing Mapped Memory Manager for %lu at %lX",
464 machine.MapMemSize, machine.MapMemBase);
465 // Create the first segment
466 void *baseAddr = reinterpret_cast<void *>(machine.MapMemBase);
467 regions[baseAddr] = {machine.MapMemSize, true};
468 }
469
480 void *get_region(uint64_t size) {
481 UPDOWN_INFOMSG("Allocating new region of size %lu bytes", size);
482 // Iterate over the regions finding one that fits
483 auto used_reg = regions.end();
484 for (auto it = regions.begin(); it != regions.end(); ++it) {
485 // Check if region is free and have enough size
486 if (it->second.free && size <= it->second.size) {
487 UPDOWN_INFOMSG("Found a region at %lX", reinterpret_cast<uint64_t>(it->first));
488 used_reg = it;
489 break;
490 }
491 }
492 // Check if we found space
493 if (used_reg == regions.end()) {
494 UPDOWN_ERROR("Allocator run out of memory. Cannot allocate %lu bytes",
495 size);
496 return nullptr;
497 }
498 // split the new region
499 used_reg->second.free = false;
500 uint64_t new_size = used_reg->second.size - size;
501 used_reg->second.size = size;
502 // Create a new empty region
503 if (new_size != 0) {
504 void *new_reg = static_cast<char *>(used_reg->first) + size;
505 regions[new_reg] = {new_size, true};
506 UPDOWN_INFOMSG("Creating a new region %lX with the remaining size %lu",
507 reinterpret_cast<uint64_t>(new_reg), new_size);
508 }
509 UPDOWN_INFOMSG("Returning region %lX = {%lu, %s}",
510 reinterpret_cast<uint64_t>(used_reg->first), used_reg->second.size,
511 (used_reg->second.free)? "Free":"Used");
512 // Return the new pointer
513 return used_reg->first;
514 }
515
526 void remove_region(void *ptr) {
527 UPDOWN_INFOMSG("Freeing the space at %lX", reinterpret_cast<uint64_t>(ptr));
528 // Find location to free
529 auto it = regions.find(ptr);
530 if (it == regions.end() || it->second.free) {
531 UPDOWN_ERROR("Trying to free pointer %lX that is not in the regions"
532 " or the region is free (double free?)",
533 reinterpret_cast<uint64_t>(ptr));
534 return;
535 }
536 // merge left if free
537 if (it != regions.begin() && std::prev(it, 1)->second.free) {
538 uint64_t size = it->second.size;
539 it--;
540 it->second.size += size;
541 UPDOWN_INFOMSG("Merging left %lX to %lX, adding %lu for a total of region with %lu",
542 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(std::next(it, 1)->first),
543 size, it->second.size);
544 UPDOWN_INFOMSG("Removing previous region at %lX",
545 reinterpret_cast<uint64_t>(it->first));
546 regions.erase(std::next(it, 1));
547
548 }
549 // merge right if free
550 auto nextIt = std::next(it, 1);
551 if (nextIt != regions.end() && nextIt->second.free) {
552 uint64_t size = nextIt->second.size;
553 it->second.size += size;
554 UPDOWN_INFOMSG("Merging right %lX to %lX, adding %lu for a total of region with %lu",
555 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(nextIt->first),
556 size, it->second.size);
557 UPDOWN_INFOMSG("Removing previous region at %lX",
558 reinterpret_cast<uint64_t>(it->first));
559 regions.erase(nextIt);
560 }
561
562 // Check if there were no merges
563 if (it->first == ptr) {
564 UPDOWN_INFOMSG("No merges performed, just freeing %lX = {%lu, %s}",
565 reinterpret_cast<uint64_t>(it->first), (it->second.size),
566 (it->second.free)? "Free":"Used");
567 it->second.free = true;
568 }
569 }

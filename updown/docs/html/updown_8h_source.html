220private:
221 uint8_t UdId; // UpDown ID

237 EventWord(((LaneId << 24) & 0xff000000) |
238 ((ThreadId << 16) & 0xff0000) | ((0 << 8) & 0xff00) |
239 (EventLabel & 0xff)) {
240 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "

257 Operands(operands),
258 EventWord(
259 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
260 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
261 0xff00) |
262 (EventLabel & 0xff)) {
263 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
264 "thread_id = %d, num_operands = %d, ev_word = %X",
265 EventLabel, LaneId, ThreadId,
266 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
267 EventWord);

278 void set_event(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
279 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr) {
280 EventLabel = e_label;
281 LaneId = lid;
282 UdId = udid;
283 ThreadId = tid;
284 Operands = operands;
285 EventWord =
286 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
287 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
288 0xff00) |
289 (EventLabel & 0xff);
290 UPDOWN_INFOMSG("Setting the event word = %d, lane_id = %d, "
291 "thread_id = %d, num_operands = %d, ev_word = %X",
292 EventLabel, LaneId, ThreadId,
293 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
294 EventWord);
295 }

503 class ud_mapped_memory_t {
504 private:
511 struct mem_region_t {
512 uint64_t size; // Size of the region
513 bool free; // Is the region free or being used
514 };
515
516 // Map to keep track of the free and used regions.
517 std::map<void *, mem_region_t> regions;
518
519 public:
528 ud_mapped_memory_t(ud_machine_t &machine) {
529 UPDOWN_INFOMSG("Initializing Mapped Memory Manager for %lu at %lX",
530 machine.MapMemSize, machine.MapMemBase);
531 // Create the first segment
532 void *baseAddr = reinterpret_cast<void *>(machine.MapMemBase);
533 regions[baseAddr] = {machine.MapMemSize, true};
534 }
535
546 void *get_region(uint64_t size) {
547 UPDOWN_INFOMSG("Allocating new region of size %lu bytes", size);
548 // Iterate over the regions finding one that fits
549 auto used_reg = regions.end();
550 for (auto it = regions.begin(); it != regions.end(); ++it) {
551 // Check if region is free and have enough size
552 if (it->second.free && size <= it->second.size) {
553 UPDOWN_INFOMSG("Found a region at %lX", reinterpret_cast<uint64_t>(it->first));
554 used_reg = it;
555 break;
556 }
557 }
558 // Check if we found space
559 if (used_reg == regions.end()) {
560 UPDOWN_ERROR("Allocator run out of memory. Cannot allocate %lu bytes",
561 size);
562 return nullptr;
563 }
564 // split the new region
565 used_reg->second.free = false;
566 uint64_t new_size = used_reg->second.size - size;
567 used_reg->second.size = size;
568 // Create a new empty region
569 if (new_size != 0) {
570 void *new_reg = static_cast<char *>(used_reg->first) + size;
571 regions[new_reg] = {new_size, true};
572 UPDOWN_INFOMSG("Creating a new region %lX with the remaining size %lu",
573 reinterpret_cast<uint64_t>(new_reg), new_size);
574 }
575 UPDOWN_INFOMSG("Returning region %lX = {%lu, %s}",
576 reinterpret_cast<uint64_t>(used_reg->first), used_reg->second.size,
577 (used_reg->second.free)? "Free":"Used");
578 // Return the new pointer
579 return used_reg->first;
580 }
581
592 void remove_region(void *ptr) {
593 UPDOWN_INFOMSG("Freeing the space at %lX", reinterpret_cast<uint64_t>(ptr));
594 // Find location to free
595 auto it = regions.find(ptr);
596 if (it == regions.end() || it->second.free) {
597 UPDOWN_ERROR("Trying to free pointer %lX that is not in the regions"
598 " or the region is free (double free?)",
599 reinterpret_cast<uint64_t>(ptr));
600 return;
601 }
602 // merge left if free
603 if (it != regions.begin() && std::prev(it, 1)->second.free) {
604 uint64_t size = it->second.size;
605 it--;
606 it->second.size += size;
607 UPDOWN_INFOMSG("Merging left %lX to %lX, adding %lu for a total of region with %lu",
608 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(std::next(it, 1)->first),
609 size, it->second.size);
610 UPDOWN_INFOMSG("Removing previous region at %lX",
611 reinterpret_cast<uint64_t>(it->first));
612 regions.erase(std::next(it, 1));
613
614 }
615 // merge right if free
616 auto nextIt = std::next(it, 1);
617 if (nextIt != regions.end() && nextIt->second.free) {
618 uint64_t size = nextIt->second.size;
619 it->second.size += size;
620 UPDOWN_INFOMSG("Merging right %lX to %lX, adding %lu for a total of region with %lu",
621 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(nextIt->first),
622 size, it->second.size);
623 UPDOWN_INFOMSG("Removing previous region at %lX",
624 reinterpret_cast<uint64_t>(it->first));
625 regions.erase(nextIt);
626 }
627
628 // Check if there were no merges
629 if (it->first == ptr) {
630 UPDOWN_INFOMSG("No merges performed, just freeing %lX = {%lu, %s}",
631 reinterpret_cast<uint64_t>(it->first), (it->second.size),
632 (it->second.free)? "Free":"Used");
633 it->second.free = true;
634 }
635 }
636 };

223private:
224 uint8_t UdId; // UpDown ID

238 event_t()

240 EventWord(((LaneId << 24) & 0xff000000) |
241 ((ThreadId << 16) & 0xff0000) | ((0 << 8) & 0xff00) |
242 (EventLabel & 0xff)) {
243 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "

257 event_t(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
258 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr)
259 : UdId(udid), LaneId(lid), ThreadId(tid), EventLabel(e_label),
260 Operands(operands),
261 EventWord(
262 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
263 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
264 0xff00) |
265 (EventLabel & 0xff)) {
266 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
267 "thread_id = %d, num_operands = %d, ev_word = %X",
268 EventLabel, LaneId, ThreadId,
269 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
270 EventWord);

281 void set_event(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
282 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr) {
283 EventLabel = e_label;
284 LaneId = lid;
285 UdId = udid;
286 ThreadId = tid;
287 Operands = operands;
288 EventWord =
289 ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
290 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) &
291 0xff00) |
292 (EventLabel & 0xff);
293 UPDOWN_INFOMSG("Setting the event word = %d, lane_id = %d, "
294 "thread_id = %d, num_operands = %d, ev_word = %X",
295 EventLabel, LaneId, ThreadId,
296 (Operands != nullptr) ? Operands->get_NumOperands() : 0,
297 EventWord);
298 }

506 class ud_mapped_memory_t {
507 private:
514 struct mem_region_t {
515 uint64_t size; // Size of the region
516 bool free; // Is the region free or being used
517 };
518
519 // Map to keep track of the free and used regions.
520 std::map<void *, mem_region_t> regions;
521
522 public:
531 ud_mapped_memory_t(ud_machine_t &machine) {
532 UPDOWN_INFOMSG("Initializing Mapped Memory Manager for %lu at %lX",
533 machine.MapMemSize, machine.MapMemBase);
534 // Create the first segment
535 void *baseAddr = reinterpret_cast<void *>(machine.MapMemBase);
536 regions[baseAddr] = {machine.MapMemSize, true};
537 }
538
549 void *get_region(uint64_t size) {
550 UPDOWN_INFOMSG("Allocating new region of size %lu bytes", size);
551 // Iterate over the regions finding one that fits
552 auto used_reg = regions.end();
553 for (auto it = regions.begin(); it != regions.end(); ++it) {
554 // Check if region is free and have enough size
555 if (it->second.free && size <= it->second.size) {
556 UPDOWN_INFOMSG("Found a region at %lX", reinterpret_cast<uint64_t>(it->first));
557 used_reg = it;
558 break;
559 }
560 }
561 // Check if we found space
562 if (used_reg == regions.end()) {
563 UPDOWN_ERROR("Allocator run out of memory. Cannot allocate %lu bytes",
564 size);
565 return nullptr;
566 }
567 // split the new region
568 used_reg->second.free = false;
569 uint64_t new_size = used_reg->second.size - size;
570 used_reg->second.size = size;
571 // Create a new empty region
572 if (new_size != 0) {
573 void *new_reg = static_cast<char *>(used_reg->first) + size;
574 regions[new_reg] = {new_size, true};
575 UPDOWN_INFOMSG("Creating a new region %lX with the remaining size %lu",
576 reinterpret_cast<uint64_t>(new_reg), new_size);
577 }
578 UPDOWN_INFOMSG("Returning region %lX = {%lu, %s}",
579 reinterpret_cast<uint64_t>(used_reg->first), used_reg->second.size,
580 (used_reg->second.free)? "Free":"Used");
581 // Return the new pointer
582 return used_reg->first;
583 }
584
595 void remove_region(void *ptr) {
596 UPDOWN_INFOMSG("Freeing the space at %lX", reinterpret_cast<uint64_t>(ptr));
597 // Find location to free
598 auto it = regions.find(ptr);
599 if (it == regions.end() || it->second.free) {
600 UPDOWN_ERROR("Trying to free pointer %lX that is not in the regions"
601 " or the region is free (double free?)",
602 reinterpret_cast<uint64_t>(ptr));
603 return;
604 }
605 // merge left if free
606 if (it != regions.begin() && std::prev(it, 1)->second.free) {
607 uint64_t size = it->second.size;
608 it--;
609 it->second.size += size;
610 UPDOWN_INFOMSG("Merging left %lX to %lX, adding %lu for a total of region with %lu",
611 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(std::next(it, 1)->first),
612 size, it->second.size);
613 UPDOWN_INFOMSG("Removing previous region at %lX",
614 reinterpret_cast<uint64_t>(it->first));
615 regions.erase(std::next(it, 1));
616
617 }
618 // merge right if free
619 auto nextIt = std::next(it, 1);
620 if (nextIt != regions.end() && nextIt->second.free) {
621 uint64_t size = nextIt->second.size;
622 it->second.size += size;
623 UPDOWN_INFOMSG("Merging right %lX to %lX, adding %lu for a total of region with %lu",
624 reinterpret_cast<uint64_t>(it->first), reinterpret_cast<uint64_t>(nextIt->first),
625 size, it->second.size);
626 UPDOWN_INFOMSG("Removing previous region at %lX",
627 reinterpret_cast<uint64_t>(it->first));
628 regions.erase(nextIt);
629 }
630
631 // Check if there were no merges
632 if (it->first == ptr) {
633 UPDOWN_INFOMSG("No merges performed, just freeing %lX = {%lu, %s}",
634 reinterpret_cast<uint64_t>(it->first), (it->second.size),
635 (it->second.free)? "Free":"Used");
636 it->second.free = true;
637 }
638 }

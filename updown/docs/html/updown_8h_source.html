153 class event_t {
154 private:
155 uint8_t UdId; // UpDown ID
156 uint8_t LaneId; // Lane ID
157 uint8_t ThreadId; // Thread ID. ANY_THREAD for any thread
158 uint8_t EventLabel; // Number representing the event label
159
160 operands_t *Operands; // Operands to be sent with this event
161
162 word_t EventWord; // Concatenated event word that contains all the fields
163
164 public:
169 event_t() :
170 UdId(0),
171 LaneId(0),
172 ThreadId(0xFF),
173 EventLabel(0),
174 Operands(nullptr),
175 EventWord(((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
176 ((0 << 8) & 0xff00) | (EventLabel & 0xff))
177 {
178 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
179 "thread_id = %d, num_operands = %d, ev_word = %X",
180 EventLabel, LaneId, ThreadId, 0, EventWord);
181 }
182
192 event_t(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
193 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr) :
194 UdId(udid),
195 LaneId(lid),
196 ThreadId(tid),
197 EventLabel(e_label),
198 Operands(operands),
199 EventWord(((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
200 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) & 0xff00) | (EventLabel & 0xff))
201 {
202 UPDOWN_INFOMSG("Creating a new event label = %d, lane_id = %d, "
203 "thread_id = %d, num_operands = %d, ev_word = %X",
204 EventLabel, LaneId, ThreadId,
205 (Operands != nullptr) ? Operands->get_NumOperands() : 0, EventWord);
206 }
207
216 void set_event(uint8_t e_label, uint8_t udid, uint8_t lid = 0,
217 uint8_t tid = ANY_THREAD, operands_t *operands = nullptr) {
218 EventLabel = e_label;
219 LaneId = lid;
220 UdId = udid;
221 ThreadId = tid;
222 Operands = operands;
223 EventWord = ((LaneId << 24) & 0xff000000) | ((ThreadId << 16) & 0xff0000) |
224 (((Operands != nullptr ? Operands->get_NumOperands() : 0) << 8) & 0xff00) | (EventLabel & 0xff);
225 UPDOWN_INFOMSG("Setting the event word = %d, lane_id = %d, "
226 "thread_id = %d, num_operands = %d, ev_word = %X",
227 EventLabel, LaneId, ThreadId, (Operands != nullptr) ? Operands->get_NumOperands() : 0, EventWord);
228 }
229
235 word_t get_EventWord() { return EventWord; }
236
237 operands_t* get_Operands() { return Operands; }
238
239 void set_operands(operands_t *ops) { Operands = ops; }
240 uint8_t get_UdId() { return UdId; }
241 uint8_t get_LaneId() { return LaneId; }
242 uint8_t get_ThreadId() { return ThreadId; }
243 uint8_t get_EventLabel() { return EventLabel; }
244 uint8_t get_NumOperands() { return (Operands != nullptr) ? Operands->get_NumOperands() : 0; }
245 ptr_t get_OperandsData() { return (Operands != nullptr) ? Operands->get_Data() : nullptr; }
246 };
